# 创建应用

# 下载依赖

    yarn init && yarn add electron --dev

# 入口文件说明

// app 控制应用事件的生命周期
// BrowserWindow 负责创建和管理应用的窗口。
const { app, BrowserWindow } = require("electron");

const createWindow = () => {
  const win = new BrowserWindow({
    width: 800,
    height: 600,
  });

  win.loadFile("index.html");
};

app.whenReady().then(() => {
  createWindow();
});
