# webpack 学习笔记

# 1、下载 webpack 以及 webpack-cli

  • 初始化 npm
npm init
  • 下载 webpack 以及 webpack-cli
npm install webpack webpack-cli --save-dev
  • 下载 webpack-dev-server
npm install webpack-dev-server --save-dev // 热更新 以及 web-server 功能  主要配置是在 devServer

# 2、webpack 配置说明 webpack.config.js

const path = require("path");
// 自动将编译html插件 npm install html-webpack-plugin
const HtmlWebpackPlugin = require("html-webpack-plugin");
// 将css抽离出来以 link 形式引入 // npm install mini-css-extract-plugin
const MiniCssExtractPlugin = require("mini-css-extract-plugin");
// 将css 进行压缩 // npm install css-minimizer-webpack-plugin
const CssMinimizerWebpackPlugin = require("css-minimizer-webpack-plugin");


module.exports = {
  entry: "./src/index.js", // 入口文件
  output: {
    filename: "bundle.js", // 打包之后的入口文件
    path: path.resolve(__dirname, "./dist"), // 打包的之后文件的名称
    clean: true, // 每次打包之前将上一次打包内容清空
    assetModuleFilename: "images/[contenthash][ext]", // 静态资源输出的位置
  },
  mode: "none", // 提供 mode 配置选项，告知 webpack 使用相应模式的内置优化。
  devtool: "inline-source-map",  // 用于调试精准定位代码
  // 插件
  plugins: [
    new HtmlWebpackPlugin({
      template: "./index.html", // 打包index.htm l入口
      filename: "app.html", // 打包之后的文件名称
      inject: "body", // bundle.js 插入的文件 不指定的话默认在 head 上面
    }),
    new MiniCssExtractPlugin({
      filename: "styles/[contenthash].css", // 打包之后css 存放的位置
    }),
  ],
  module: {
    rules: [
      {
        test: /\.png$/,
        type: "asset/resource", // 图片资源
        generator: {
          filename: "images/[contenthash][ext]", // 打包之后的地址和文件名
        },
      },
      {
        test: /\.svg$/,
        type: "asset/inline", // 打包会编译成base64 文件
      },
      {
        test: /\.txt$/, 
        type: "asset/source", // 源文件
      },
      {
        test: /\.jpeg$/,
        type: "asset", // 自动 根据文件大小自动变成成 resource 或者 inline
        parser: {
          dataUrlCondition: {
            maxSize: 4 * 1024 * 1024, // 如果 4M以内打包成inline
          },
        },
      },
    ],
  },
  devServer: {
    static: "./dist", // 该配置项允许配置从目录提供静态文件的选项（默认是 'public' 文件夹）。将其设置为 false 以禁用：
  },
};
