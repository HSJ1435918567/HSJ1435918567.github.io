# TS 学习笔记

# 一、基础类型

# 布尔类型 blooean

blooean let bol:blooean = true|false

# 数组类型 number

number let num:number = 1

# 字符串类型 string

string let str:string="好好学习"

# 元组 Tuple

元组类型允许表示一个已知元素数量和类型的数组，各元素的类型不必相同。

Tuple let tup:[string,number] = ['hello',1]

# 任意 any

any let not:any = "1" | let not:any = 1 | let not:any = {a:1} | let not:any = [1]

# Void void

function type:void(){
    console.log("huangshaojun")
}

# 不常用到

never 类型表示的是那些永不存在的值的类型

undefined

null

# 类型断言 as <>

as 使用方法:
    let str:any = "lyc大傻x"
    let strLength:number = (str as string).length
<> 使用方法:
    let str:any = "lyc大傻x"
    let strLength:number = (<string>str).length

# 二、接口

# 基本使用

interface Str {
    label:string,
    value:number
}
let str:Str = {label:'星期一'，value:1}

# 可选属性

interface Str {
    label:string
    value?:number
}
let str:Str = {label:'星期一'}

# 只读属性

一些对象属性只能在对象刚刚创建的时候修改其值。

interface Str{
    readonly label:string
}

let str:Str = {label:'星期一'} // 首次赋值不会报错
str.label = "星期二" // 报错

# 函数类型

interface SearchFunc {
  (source: string, subString: string): boolean;
}

let mySearch: SearchFunc;
mySearch = function (source: string, subString: string) {
  let result = source.search(subString);
  return result > -1;
};
