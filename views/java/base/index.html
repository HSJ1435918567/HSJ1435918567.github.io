# Java基础

# 1、java IO 流这块有四大家族

  • java.io.InputStreamjava 字节输入流

  • java.io.OutputStream 字节输出流

  • java.io.Reader 字符输入流

  • java.io.Writer 字符输出流

所有的 输出流 都应该刷新后关闭 flush()、close() 所有的流 使用完都应该关闭 close()

# 2、java.io 包下需要掌握的流有 16 个

# 2.1、文件专属

  • java.io.FileInputStream
  • java.io.FileOutputStream
  • java.io.FileReader
  • java.io.FileWriter

# 2.2、转换流（将字节流转换成字符流）

  • java.io.InputStreamReader
  • java.io.OutputStreamWriter

# 2.3、缓冲流专属

  • java.io.BufferedReader
  • java.io.BufferedWriter
  • java.io.BufferedInputStream
  • java.io.BufferedOutputStream

# 2.4、标准输出流

  • java.io.PrintWriter
  • java.io.PrintStream

# 2.5、对象专属流

  • java.io.ObjectInputStream
  • java.io.ObjectOutputStream

# 3、案例

# 3.1、FileInputStream 读文件


import java.io.FileInputStream;
import java.io.IOException;

public class FileInputStreamTest01{
    public static void main(String[] args){
        FileInputStream files = null;
        try{
            // D:\\1、周报.txt 文件路径
            files = new FileInputStream("D:\\1、周报.txt");
            int num = 0;
            byte[] bytes = new byte[1024*1024]; // 1mb
            while((num = files.read(bytes)) != -1){
                String str = new String(bytes,0,num);
                //  输出读到的文件
                System.out.println(str);
            }
        }catch(IOException e){
            e.printStackTrace();
        }finally{
            // 避免空指针异常
            if(files != null){
                try {
                    // 一定要关闭
                    files.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }

    }
}

# 3.2、FileOutputStream 写文件


import java.io.FileOutputStream;
import java.io.IOException;

public class FileOutputStreamTest01{

    public static void main(String[] args){

        FileOutputStream files = null;

        try{
            // 保存的文件地址，如果没有这个文件会新建，有的话就不新建了。
            files = new FileOutputStream("D:\\1、周报.txt");
            String str = "中国人，不骗中国人!";
            // 将字符转换为字节
            byte[] bytes = str.getBytes();
            files.write(bytes);
            // 一定要记得刷新，防止有剩余的流没有输出，俗称清空管道
            files.flush();
        }catch(IOException e){
            e.printStackTrace();
        }finally{
            // 避免空指针
            if(files != null){
                try{
                    // 一定要记得关闭
                    files.close();
                }catch(IOException e){
                    e.printStackTrace();
                }
            }

        }

    }

}

# 3.3、FileReader 读文件

import java.io.FileReader;
import java.io.IOException;

public class FileReaderTest01 {
    public static void main(String[] args) {
        FileReader files = null;
        try {
            // 文件地址
            files = new FileReader("D:\\1、周报.txt");
            int num = 0;
            // 字符流
            char[] chars = new char[1028 * 1028]; // 2mb
            while ((num = files.read(chars)) != -1) {
                String str = new String(chars, 0, num);
                System.out.println(str);
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                files.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

}

# 3.4、FileWriter 写文件


import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;


public class FileWriterTest01 {

    public static void main(String[] args) {
        FileReader fileR = null;
        FileWriter fileW = null;
        try {
            // 读的地址
            fileR = new FileReader("D:\\1、周报.txt");
            // 写的地址 true 表示不清空，叠加
            fileW = new FileWriter("text.txt",true);
            // 字符流
            char[] chars = new char[1024*1024];
            int num = 0;
            while ((num = fileR.read(chars)) != -1){
                fileW.write(chars,0,num);
            }
            // 刷新，防止有剩余的流没有输出，清空管道
            fileW.flush();

        } catch (IOException e) {
            e.printStackTrace();
        }finally {
            try {
                if(fileR != null){
                    fileR.close();
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
            try {
                if(fileW != null){
                    fileW.close();
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }

    }
}

# 4、java.io.File

  • File 类和四大家族没有什么关系，所有 File 类不能完成文件的读和写。
  • File 对象代表什么？

    2. 一个 File 对象有可能对应的是目录，也可能文件
    3. File 只是一个路径名的抽象表示形式
